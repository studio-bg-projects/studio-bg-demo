UPDATE `productSpecifications` AS `ps`
JOIN `specifications` AS `s` ON (`ps`.`specificationId` = `s`.`id`)
SET `ps`.`specificationValue` = CONCAT(REGEXP_REPLACE(CONCAT(' ', `s`.`name`), ' ([^ ])[^ ]*', '$1'), '-', FLOOR(RAND() * (3 - 1 + 1)) + 1)
